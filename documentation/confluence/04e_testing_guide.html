Running Tests

# Activate the environment first
conda activate txr_automation

# Run the full suite
pytest

# Verbose output (show individual test names)
pytest -v

# Run a specific test file
pytest tests/test_accuracy_testing/test_buyer_id_validation.py

# Run a specific test by name pattern
pytest -k "test_validate_nidn_valid_uk_nino"

# Run tests matching multiple patterns
pytest -k "buyer and nidn"

# Run only fast unit tests (skip integration)
pytest -m "not integration"

# Run with coverage report
pytest --cov=src --cov-report=html

# Run with coverage and show missing lines
pytest --cov=src --cov-report=term-missing

# Run in parallel (if pytest-xdist installed)
pytest -n auto

# Stop on first failure
pytest -x

# Show local variables in tracebacks
pytest -l

Fixtures & conftest.py

Root conftest.py

Located at tests/conftest.py, it adds the project root to sys.path for src.* imports:

import sys
from pathlib import Path

project_root = Path(__file__).parent.parent
if str(project_root) not in sys.path:
    sys.path.insert(0, str(project_root))

Writing New Tests

Step-by-Step

  1. Identify the module under test — find the matching test_ file or create one
  2. Write the test function with a descriptive name
  3. Add a docstring explaining what is being tested and why
  4. Use fixtures for shared setup (data, config, temp files)
  5. Assert specific outcomes — one concept per test
  6. Run the test: pytest tests/path/to/test_file.py -v

Template

"""Tests for the XYZ validation module."""

import pytest
from src.accuracy_testing.validators.xyz_validator import XYZValidator


class TestXYZValidator:
    """Tests for XYZValidator class."""

    def test_validate_valid_input_returns_true(self):
        """Validate that a correct input passes validation."""
        validator = XYZValidator()
        result = validator.validate("VALID_INPUT")
        assert result.is_valid is True
        assert len(result.errors) == 0

    def test_validate_empty_input_returns_error(self):
        """Validate that empty input is rejected with appropriate error."""
        validator = XYZValidator()
        result = validator.validate("")
        assert result.is_valid is False
        assert "empty" in result.errors[0].lower()

    @pytest.mark.parametrize("input_value,expected", [
        ("AB123456C", True),   # Valid UK NINO
        ("INVALID", False),     # Too short
        ("AB000000Z", False),   # Invalid suffix
    ])
    def test_validate_parametrized_cases(self, input_value, expected):
        """Validate multiple input/output combinations."""
        validator = XYZValidator()
        result = validator.validate(input_value)
        assert result.is_valid is expected

Testing DataFrames

import pandas as pd
from pandas.testing import assert_frame_equal

def test_processor_adds_error_column(sample_data):
    """Verify processor adds Error column to output."""
    result = processor.process(sample_data)
    assert "Error" in result.columns
    assert result["Error"].dtype == object

def test_output_matches_expected(sample_data, expected_output):
    """Compare processor output against expected results."""
    result = processor.process(sample_data)
    assert_frame_equal(result, expected_output, check_dtype=False)

def test_missing_config_raises_file_not_found():
    """Verify FileNotFoundError for missing config file."""

        AccuracyConfigManager.load_from_yaml("nonexistent.yaml")

Integration Tests

Integration tests live in tests/integration/ and test end-to-end workflows:

# Mark integration tests
@pytest.mark.integration
def test_end_to_end_buyer_validation(tmp_path):
    """Full buyer ID validation pipeline: CSV in → validated CSV out."""
    # 1. Create input CSV
    input_csv = create_test_csv(tmp_path / "input.csv")
    
    # 2. Create config
    config = create_test_config(input_csv, tmp_path / "output.csv")
    
    # 3. Run validation
    exit_code = run_buyer_validation(config)
    
    # 4. Verify output
    assert exit_code == 0
    output = pd.read_csv(tmp_path / "output.csv")
    assert "Error" in output.columns
    assert len(output) == expected_record_count
